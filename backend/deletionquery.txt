DELETE DupRows.*
FROM viewcountsbymonth AS DupRows
   INNER JOIN (
      SELECT MIN(row_id) AS minId,pagename,language,monthfull
      FROM viewcountsbymonth
      GROUP BY pagename,language,monthfull
      HAVING COUNT(*) > 1
   ) AS SaveRows ON SaveRows.pagename = DupRows.pagename AND SaveRows.language = DupRows.language AND SaveRows.monthfull = DupRows.monthfull
      AND SaveRows.minId <> DupRows.row_id;
